\section{Algorithms}

\begin{algorithm}[H]
  \caption{IsClosed Function}
  \KwData{$S, SA$}
  \KwResult{True if OT is closed, else $w \in SA | row(w) \notin upper(OT)$}
  \For{$\forall x \in SA$}{
    \If{$\nexists s \in S \text{ such that } row[x] = row[s]$}{
      \Return $\{isClosed = False, rowToPromote = x\}$;
    }
  }
  \Return{$\{isClosed = True, rowToPromote = \emptyset\}$}
\end{algorithm}

\begin{algorithm}[H]
  \caption{IsConsistent Function}
  \KwData{$S, SA, E, OT, \Sigma$}
  \KwResult{True if OT is constistent, else a column $e \in E$ to differenciate two rows}
  \For{$\forall s_1, s_2 \in S$}{
    \If{$OT[s_1] = OT[s_2]$}{
      \For{$\forall x \in \Sigma, \forall e \in E$}{
        \If{$OT[s_1 \cdot x \cdot e] \neq OT[s_2 \cdot x \cdot e]$}{
          \Return{$\{isConsistent = False, rowToAdd = x \cdot e\}$}
        }
      }
    }
  }
  \Return{$\{isConsistent = True, columnToAdd = \emptyset\}$}
\end{algorithm}
